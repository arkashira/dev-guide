from django.db import models
from django.core.mail import send_mail
from .models import Mentor

class MentorshipRequest(models.Model):
    """Represents a mentorship request form data"""
    user = models.ForeignKey('auth.User', on_delete=models.CASCADE)
    goals = models.TextField()
    availability = models.TextField()
    created_at = models.DateTimeField(auto_now_add=True)

    def match_mentor(self):
        """Matches the request to a mentor with complementary expertise"""
        mentors = Mentor.objects.all()
        best_match = None
        best_match_score = 0

        for mentor in mentors:
            # Calculate the match score based on the mentor's expertise and the user's goals
            match_score = calculate_match_score(self.goals, mentor.expertise)

            if match_score > best_match_score:
                best_match = mentor
                best_match_score = match_score

        return best_match

    def send_notification_emails(self, mentor):
        """Sends notification emails to both parties"""
        user_email = self.user.email
        mentor_email = mentor.email

        send_mail(
            'Mentorship Request Matched',
            f'Dear {self.user.username}, your mentorship request has been matched with {mentor.name}.',
            'dev-guide@example.com',
            [user_email],
            fail_silently=False,
        )

        send_mail(
            'Mentorship Request Received',
            f'Dear {mentor.name}, you have been matched with {self.user.username} for mentorship.',
            'dev-guide@example.com',
            [mentor_email],
            fail_silently=False,
        )

def calculate_match_score(goals, expertise):
    """Calculates the match score based on the user's goals and the mentor's expertise"""
    # Implement a simple keyword matching algorithm for demonstration purposes
    goals_keywords = [keyword.strip() for keyword in goals.split(',')]
    expertise_keywords = [keyword.strip() for keyword in expertise.split(',')]

    match_score = len(set(goals_keywords) & set(expertise_keywords))

    return match_score
