import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class PersonalizedLearningRecommendation:
    def __init__(self, data_path):
        self.data = pd.read_csv(data_path)
        self.vectorizer = TfidfVectorizer()
        self.tfidf_matrix = None
        self.similarity_matrix = None

    def preprocess_data(self):
        self.data['content'] = self.data['title'] + ' ' + self.data['description']
        self.tfidf_matrix = self.vectorizer.fit_transform(self.data['content'])
        self.similarity_matrix = cosine_similarity(self.tfidf_matrix, self.tfidf_matrix)

    def get_recommendations(self, user_id, top_n=5):
        user_data = self.data[self.data['user_id'] == user_id]
        if user_data.empty:
            return []

        user_index = user_data.index[0]
        similarity_scores = list(enumerate(self.similarity_matrix[user_index]))
        similarity_scores = sorted(similarity_scores, key=lambda x: x[1], reverse=True)
        top_indices = [i[0] for i in similarity_scores[1:top_n+1]]
        return self.data.iloc[top_indices]

# Example usage
if __name__ == "__main__":

    recommender.preprocess_data()

- Progress updates automatically when a course is marked complete via API call.
- User can view a detailed progress report modal with a breakdown of completed vs pending items.
- Progress bars show visual indication of completion status (color coding).
- Modal displays total courses, completed courses, and completion percentage.
- Component handles loading states gracefully during data fetch.
- Progress bars are responsive and maintain correct display on different screen sizes.

describe('ProgressBar', () => {
  it('renders progress bar with correct width based on percentage', () => {
    const wrapper = shallowMount(ProgressBar, {
      propsData: { percentage: 75 }
    })
    expect(wrapper.find('.progress-bar').attributes('style')).toContain('width: 75%')
  })

  it('shows appropriate color class based on completion percentage', () => {
    const wrapper = shallowMount(ProgressBar, { propsData: { percentage: 30 } })
    expect(wrapper.classes()).toContain('progress-low')
    
    const wrapper2 = shallowMount(ProgressBar, { propsData: { percentage: 60 } })
    expect(wrapper2.classes()).toContain('progress-medium')
    
    const wrapper3 = shallowMount(ProgressBar, { propsData: { percentage: 90 } })
    expect(wrapper3.classes()).toContain('progress-high')
  })

  it('displays percentage text correctly', () => {
    const wrapper = shallowMount(ProgressBar, { propsData: { percentage: 45 } })
    expect(wrapper.find('.percentage-text').text()).toBe('45%')
  })

  it('handles loading state correctly', () => {
    const wrapper = shallowMount(ProgressBar, { propsData: { percentage: null, loading: true } })
    expect(wrapper.find('.loading-spinner').exists()).toBe(true)
  })

// Dashboard.vue tests
describe('Dashboard', () => {
  it('fetches and displays recommendations with progress bars', async () => {
    const mockRecommendations = [
      { id: 1, title: 'Course 1', progress: 75 },
      { id: 2, title: 'Course 2', progress: 30 }
    ]
    
    const wrapper = shallowMount(Dashboard, {
      mocks: {
        $http: {
          get: jest.fn().mockResolvedValue(mockRecommendations)
        }
      }
    })
    
    await wrapper.vm.$nextTick()
    expect(wrapper.findAllComponents(ProgressBar)).toHaveLength(2)
  })

  it('updates progress when course completion is marked', async () => {
    const mockRecommendations = [{ id: 1, title: 'Course 1', progress: 0 }]
    
    const wrapper = shallowMount(Dashboard, {
      mocks: {
        $http: {
          get: jest.fn().mockResolvedValue(mockRecommendations),
          post: jest.fn().mockResolvedValue({ success: true })
        }
      }
    })
    
    await wrapper.vm.$nextTick()
    
    // Simulate API call to mark course complete
    await wrapper.vm.markCourseComplete(1)
    expect(wrapper.vm.recommendations[0].progress).toBe(100) // Assuming completion sets progress to 100%
  })

  it('displays progress report modal correctly', async () => {
    const mockRecommendations = [
      { id: 1, title: 'Course 1', progress: 75 },
      { id: 2, title: 'Course 2', progress: 30 }
    ]
    
    const wrapper = shallowMount(Dashboard, {
      mocks: {
        $http: {
          get: jest.fn().mockResolvedValue(mockRecommendations)
        }
      }
    })
    
    await wrapper.vm.$nextTick()
    wrapper.vm.showProgressReport() // Trigger modal display
    expect(wrapper.find('.progress-report-modal').isVisible()).toBe(true)
    expect(wrapper.find('.total-courses').text()).toContain('Total Courses: 2')
    expect(wrapper.find('.completed-courses').text()).toContain('Completed Courses: 1')
    expect(wrapper.find('.completion-percentage').text()).toContain('Completion Percentage: 75%')
  })
})
