<template>
  <div class="progress-bar" :data-testid="`progress-${id}`">
    <div class="progress-bar__label">
      <span class="progress-bar__title">{{ title }}</span>
      <span class="progress-bar__pct">{{ percent }}%</span>
    </div>
    <div
      class="progress-bar__track"
      role="progressbar"
      :aria-valuenow="percent"
      aria-valuemin="0"
      aria-valuemax="100"
      :aria-label="`Progress for ${title}`"
    >
      <div
        class="progress-bar__fill"
        :class="{ 'is-complete': percent >= 100 }"
        :style="{ width: `${percent}%` }"
      />
    </div>
  </div>
</template>

<script setup>
import { computed, watch } from 'vue'

const props = defineProps({
  id: { type: [String, Number], required: true },
  title: { type: String, required: true },
  completed: { type: Number, default: 0 },
  total: { type: Number, default: 1 }
})

const emit = defineEmits(['completed'])

const percent = computed(() => {
  if (!props.total) return 0
  const p = Math.round((props.completed / props.total) * 100)
  return Math.min(100, Math.max(0, p))
})

watch(percent, (v, prev) => {
  if (v === 100 && prev !== 100) emit('completed', props.id)
})
</script>

<style scoped>
/* Styles remain unchanged */
</style>
```
