### 1.1 `src/components/MentorshipRequestForm.vue`

```vue
<template>
  <form @submit.prevent="submitMentorshipRequest" class="mentorship-form">
    <h2>Request a Mentor</h2>

    <label for="goals">Goals (max 500 chars)</label>
    <textarea
      id="goals"
      v-model="form.goals"
      rows="4"
      maxlength="500"
      required
    ></textarea>

    <label for="availability">Availability (e.g., Mon‑Wed 10‑12)</label>
    <input
      id="availability"
      v-model="form.availability"
      type="text"
      required
    />

    <label for="expertise">Desired Expertise</label>
    <input
      id="expertise"
      v-model="form.expertise"
      type="text"
      placeholder="e.g., Front‑end, Data Science"
      required
    />

    <button type="submit" :disabled="submitting">
      {{ submitting ? 'Submitting…' : 'Submit Request' }}
    </button>
  </form>
</template>

<script>
export default {
  data() {
    return {
      form: {
        goals: '',
        availability: '',
        expertise: '',
      },
      submitting: false,
    };
  },
  methods: {
    async submitMentorshipRequest() {
      this.submitting = true;
      try {
        const payload = { ...this.form, userId: this.$store.state.user.id };
        const res = await this.$axios.post('/api/mentorship-requests', payload);
        this.$toast.success(res.data.message);
        this.form = { goals: '', availability: '', expertise: '' };
      } catch (err) {
        const msg = err.response?.data?.error || 'Submission failed';
        this.$toast.error(msg);
      } finally {
        this.submitting = false;
      }
    },
  },
};
</script>
```
