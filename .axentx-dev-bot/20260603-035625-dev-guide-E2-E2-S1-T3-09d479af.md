<template>
  <div class="mentorship-request-form">
    <h2>Request a Mentor</h2>
    <form @submit.prevent="submitForm" novalidate>
      <div class="form-group">
        <label for="goals">Your Goals</label>
        <textarea
          id="goals"
          v-model="form.goals"
          rows="4"
          required
          :class="{ invalid: errors.goals }"
        />
        <span v-if="errors.goals" class="error">{{ errors.goals }}</span>
      </div>

      <div class="form-group">
        <label for="availability">Availability</label>
        <input
          id="availability"
          type="datetime-local"
          v-model="form.availability"
          required
          :class="{ invalid: errors.availability }"
        />
        <span v-if="errors.availability" class="error">{{ errors.availability }}</span>
      </div>

      <button type="submit" :disabled="submitting">
        {{ submitting ? 'Submitting…' : 'Submit Request' }}
      </button>
    </form>
  </div>
</template>

<script setup>
import { reactive, ref } from 'vue';
import axios from 'axios';
import { useToast } from 'vue-toastification';

const toast = useToast();

const form = reactive({
  goals: '',
  availability: '',
});

const errors = reactive({
  goals: '',
  availability: '',
});

const submitting = ref(false);

function validate() {
  let valid = true;
  errors.goals = '';
  errors.availability = '';

  if (!form.goals.trim()) {
    errors.goals = 'Goals are required.';
    valid = false;
  }
  if (!form.availability) {
    errors.availability = 'Availability is required.';
    valid = false;
  }
  return valid;
}

async function submitForm() {
  if (!validate()) return;
  submitting.value = true;
  try {
    await axios.post('/api/mentorship-requests', {
      goals: form.goals,
      availability: form.availability,
    });
    toast.success('Mentorship request submitted successfully!');
    form.goals = '';
    form.availability = '';
  } catch (err) {
    toast.error('Failed to submit request. Please try again.');
  } finally {
    submitting.value = false;
  }
}
</script>

## 1. Acceptance Criteria

- User can navigate to the mentorship page and see a request form with at minimum a "goals" text field and an "availability" selector/input.
- Form validates required fields client-side; submission is blocked and errors are shown if either field is empty.
- Submitting valid form data calls `POST /api/mentorships/request` and receives a 201 response containing a `matchedMentor` object.
- The matched mentor has at least one expertise tag that semantically overlaps the user's stated goals.
- After a successful match, the requesting user receives a notification email containing the mentor's name and contact info.
- After a successful match, the matched mentor receives a notification email containing the mentee's name and goals.
- If no mentor is available, the API returns a 202 (queued) status and no email is sent until a match is found.
