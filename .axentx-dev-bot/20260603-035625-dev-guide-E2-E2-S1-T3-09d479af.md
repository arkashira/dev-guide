// frontend/src/services/mentorship.ts
import axios from 'axios';

const http = axios.create({ baseURL: '/api/mentorship' });

export interface MentorshipRequest {
  goals: string;
  availability: string[];          // e.g. ["weekdays-evening", "weekends"]
  experienceLevel: 'junior' | 'mid' | 'senior';
  preferredLanguage?: string;      // programming language / stack
}

export interface MentorMatch {
  id: string;
  name: string;
  headline: string;
  expertise: string[];
  matchScore: number;
}

export interface MentorshipRequestResponse {
  requestId: string;
  match: MentorMatch;
  notificationStatus: {
    requester: 'sent' | 'pending';
    mentor: 'sent' | 'pending';
  };
}

export async function submitMentorshipRequest(
  payload: MentorshipRequest,
): Promise<MentorshipRequestResponse> {
  const { data } = await http.post<MentorshipRequestResponse>('/requests', payload);
  return data;
}
```
