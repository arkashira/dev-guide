```js
// RecommendationWidget.spec.js
import { shallowMount, createLocalVue } from '@vue/test-utils'
import RecommendationWidget from '@/components/RecommendationWidget.vue'
import axios from 'axios'
import Vuex from 'vuex'
import VueRouter from 'vue-router'

jest.mock('axios')

const localVue = createLocalVue()
localVue.use(Vuex)
localVue.use(VueRouter)

describe('RecommendationWidget.vue', () => {
  let store
  let router

  beforeEach(() => {
    store = new Vuex.Store({
      state: {
        user: {
          skills: ['JavaScript', 'Vue'],
          careerGoals: ['Frontend Engineer']
        }
      }
    })
    router = new VueRouter()
  })

  it('fetches recommendations on mount with correct params', async () => {
    axios.get.mockResolvedValue({ data: [{ id: 1, title: 'Vue 3 Basics' }] })
    const wrapper = shallowMount(RecommendationWidget, { store, localVue })
    await wrapper.vm.$nextTick()
    expect(axios.get).toHaveBeenCalledWith(
      '/api/recommendations',
      expect.objectContaining({
        params: {
          skills: 'JavaScript,Vue',
          goals: 'Frontend Engineer'
        }
      })
    )
  })

  it('renders at least 3 recommendation cards', async () => {
    axios.get.mockResolvedValue({
      data: [
        { id: 1, title: 'Vue 3 Basics', shortDescription: 'Learn Vue 3', thumbnail: 'thumb1.png' },
        { id: 2, title: 'Advanced Vue', shortDescription: 'Deep dive', thumbnail: 'thumb2.png' },
        { id: 3, title: 'Vue Router', shortDescription: 'Routing', thumbnail: 'thumb3.png' }
      ]
    })
    const wrapper = shallowMount(RecommendationWidget, { store, localVue })
    await wrapper.vm.$nextTick()
    expect(wrapper.findAll('.recommendation-card').length).toBe(3)
  })

  it('navigates to course detail on card click', async () => {
    axios.get.mockResolvedValue({ data: [{ id: 42, title: 'Test Course' }] })
    const wrapper = shallowMount(RecommendationWidget, { store, localVue, router })
    await wrapper.vm.$nextTick()
    await wrapper.find('.recommendation-card').trigger('click')
    expect(wrapper.vm.$router.currentRoute.path).toBe('/courses/42')
  })

  it('shows error message when API fails', async () => {
    axios.get.mockRejectedValue(new Error('Network Error'))
    const wrapper = shallowMount(RecommendationWidget, { store, localVue })
    await wrapper.vm.$nextTick()
    expect(wrapper.text()).toContain('Unable to load recommendations')
  })
})
```
