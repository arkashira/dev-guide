from typing import List, Dict


class DevGuide:
    """
    Core logic for the DevGuide feature:
    - Generates personalized learning recommendations based on user interests and experience.
    - Matches users with mentors based on skills and development goals.
    """

    # Sample data sources (in a real system these would be fetched from a database or external service)
    _learning_resources = [
        {"title": "Intro to AI", "tags": ["AI", "Beginner"]},
        {"title": "Advanced Machine Learning", "tags": ["AI", "Machine Learning", "Advanced"]},
        {"title": "Web Development with Python", "tags": ["Python", "Web Development", "Intermediate"]},
        {"title": "JavaScript for Front‑End", "tags": ["JavaScript", "Web Development", "Beginner"]},
        {"title": "Data Engineering Fundamentals", "tags": ["Data Engineering", "Intermediate"]},
    ]

    _mentors = [
        {"name": "Alice", "skills": ["Python", "Machine Learning"], "available": True},
        {"name": "Bob", "skills": ["JavaScript", "Web Development"], "available": True},
        {"name": "Carol", "skills": ["Data Engineering", "SQL"], "available": False},
        {"name": "Dave", "skills": ["AI", "Deep Learning"], "available": True},
    ]

    def get_personalized_recommendations(self, user_data: Dict) -> List[Dict]:
        """
        Return a list of learning resources that match the user's interests.
        Simple relevance algorithm:
          - Include a resource if any of its tags intersect with the user's interests.
          - If the user has >5 years experience, also include resources tagged as "Advanced".
        """
        interests = set(user_data.get("interests", []))
        experience = user_data.get("experience", 0)

        recommendations = []
        for resource in self._learning_resources:
            tags = set(resource["tags"])
            if interests & tags:
                # Basic match
                recommendations.append(resource)
            elif experience > 5 and "Advanced" in tags:
                # Experience‑based uplift
                recommendations.append(resource)

        # De‑duplicate while preserving order
        seen = set()
        unique_recommendations = []
        for r in recommendations:
            key = r["title"]
            if key not in seen:
                seen.add(key)
                unique_recommendations.append(r)

        return unique_recommendations

    def find_mentors(self, user_data: Dict) -> List[Dict]:
        """
        Return a list of mentors whose skill set overlaps with the user's skills
        and who are currently available.
        """
        user_skills = set(user_data.get("skills", []))
        mentors = [
            mentor for mentor in self._mentors
            if mentor["available"] and user_skills.intersection(set(mentor["skills"]))
        ]

        # Sort mentors by number of matching skills (descending)
        mentors.sort(key=lambda m: len(user_skills.intersection(set(m["skills"]))), reverse=True)
        return mentors
