+from flask import Blueprint, jsonify, request
+from models import Recommendation, CourseCompletion, db
+
+progress_api = Blueprint('progress_api', __name__)
+
+# Endpoint to get progress data for all active recommendations
+@progress_api.route('/api/progress', methods=['GET'])
+def get_progress():
+    user_id = request.args.get('user_id')
+    recommendations = Recommendation.query.filter_by(user_id=user_id, status='active').all()
+    progress_data = []
+
+    for rec in recommendations:
+        completed_courses = CourseCompletion.query.filter_by(user_id=user_id, recommendation_id=rec.id).count()
+        total_courses = len(rec.courses)
+        progress_percentage = (completed_courses / total_courses) * 100 if total_courses > 0 else 0
+        progress_data.append({
+            'recommendation_id': rec.id,
+            'title': rec.title,
+            'progress': progress_percentage,
+            'completed_courses': completed_courses,
+            'total_courses': total_courses
+        })
+
+    return jsonify(progress_data)
+
+# Endpoint to mark a course as complete and update progress
+@progress_api.route('/api/progress/mark_complete', methods=['POST'])
+def mark_course_complete():
+    data = request.json
+    user_id = data.get('user_id')
+    recommendation_id = data.get('recommendation_id')
+    course_id = data.get('course_id')
+
+    completion = CourseCompletion(user_id=user_id, recommendation_id=recommendation_id, course_id=course_id)
+    db.session.add(completion)
+    db.session.commit()
+
+    return jsonify({'message': 'Course marked as complete'}), 200
```
