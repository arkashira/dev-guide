<template>
  <div class="skill-analyzer">
    <h2>Skill Gap Analysis</h2>
    <div class="input-section">
      <textarea v-model="userProfile" placeholder="Paste your resume or profile here"></textarea>
      <textarea v-model="jobRequirements" placeholder="Paste job description here"></textarea>
      <button @click="analyze">Analyze Skill Gap</button>
    </div>
    
    <div v-if="analysisResult" class="results">
      <h3>Analysis Results</h3>
      <div class="gaps">
        <h4>Skills Missing:</h4>
        <ul>
          <li v-for="gap in analysisResult.gaps" :key="gap.skill">
            {{ gap.skill }}: Current {{ gap.level }} → Required {{ gap.required }}
          </li>
        </ul>
      </div>
      <div class="recommendations">
        <h4>Recommendations:</h4>
        <ul>
          <li v-for="rec in analysisResult.recommendations" :key="rec">{{ rec }}</li>
        </ul>
      </div>
      <div class="confidence">
        Confidence: {{ Math.round(analysisResult.confidence * 100) }}%
      </div>
    </div>
  </div>
</template>

<script>
import { aiService } from '@/services/aiService';

export default {
  name: 'SkillAnalyzer',
  data() {
    return {
      userProfile: '',
      jobRequirements: '',
      analysisResult: null
    };
  },
  methods: {
    async analyze() {
      if (!this.userProfile || !this.jobRequirements) return;
      
      this.analysisResult = await aiService.analyzeSkillGap(
        this.userProfile,
        this.jobRequirements
      );
    }
  }
};
</script>
