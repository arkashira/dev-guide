## 1. Acceptance Criteria  

| # | Criterion | Measurable Target |
|---|-----------|-------------------|
| 1 | **Form Submission** | A POST to `/mentorship/request` with JSON `{userId, goals, availability}` returns HTTP 201 and stores a request record. |
| 2 | **Rule‑Based Matching** | The matcher selects a mentor whose `expertise` tags intersect with the request’s `goals` tags **and** whose `availability` overlaps with the requester’s. |
| 3 | **Single Best Match** | If multiple mentors qualify, the one with the highest tag overlap score is chosen. |
| 4 | **Notification** | Both requester and mentor receive an email containing the match details (mentor name, expertise, scheduled time). |
| 5 | **Idempotency** | Submitting the same request twice within 24 h does not create duplicate records or send duplicate emails. |
| 6 | **Error Handling** | Invalid payloads return HTTP 400 with a clear error message. |
| 7 | **Performance** | Matching completes in < 200 ms for up to 500 mentors in memory. |

import pytest
from services.mentorship_matcher import match_mentor, Mentor, Request

@pytest.fixture
def mentors():
    return [
        Mentor(id=1, name="Alice", expertise={"python", "ml"}, availability=[("Mon", 9, 12)]),
        Mentor(id=2, name="Bob", expertise={"js", "react"}, availability=[("Tue", 14, 18)]),
        Mentor(id=3, name="Carol", expertise={"python", "data"}, availability=[("Mon", 10, 13)]),
    ]

def test_exact_tag_match(mentors):
    req = Request(user_id=42, goals={"python"}, availability=[("Mon", 10, 11)])
    mentor = match_mentor(req, mentors)
    assert mentor.id == 3  # Carol has 1 tag overlap and overlapping slot

def test_multiple_overlap_scores(mentors):
    req = Request(user_id=42, goals={"python", "data"}, availability=[("Mon", 10, 11)])
    mentor = match_mentor(req, mentors)
    assert mentor.id == 3  # Carol has 2 tags vs Alice's 1

def test_no_available_mentor(mentors):
    req = Request(user_id=42, goals={"python"}, availability=[("Wed", 10, 11)])
    mentor = match_mentor(req, mentors)
    assert mentor is None

def test_invalid_request():
    with pytest.raises(ValueError):
        match_mentor(None, [])

def test_idempotent_submission(monkeypatch):
    # Simulate storage and email service
    storage = {}
    emails_sent = []

    def mock_store(req):
        key = (req.user_id, tuple(req.goals))
        if key in storage:
            raise ValueError("Duplicate")
        storage[key] = req

    def mock_email(recipient, subject, body):
        emails_sent.append((recipient, subject, body))

    monkeypatch.setattr("services.mentorship_matcher.store_request", mock_store)
    monkeypatch.setattr("services.mentorship_matcher.send_email", mock_email)

    req = Request(user_id=42, goals={"python"}, availability=[("Mon", 10, 11)])
    # First submission
    services.mentorship_matcher.handle_request(req)
    assert len(emails_sent) == 2  # requester + mentor

    # Second submission within 24h
    with pytest.raises(ValueError):
        services.mentorship_matcher.handle_request(req)
    assert len(emails_sent) == 2  # no new emails
